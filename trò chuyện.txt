<script>
    // 1. Xử lý Đăng nhập
    function handleLogin() {
        const user = document.getElementById('username').value;
        if (user.trim() !== "") {
            localStorage.setItem('club_user', user);
            document.getElementById('login-screen').classList.add('hidden');
            document.getElementById('coach-screen').classList.remove('hidden');
            document.getElementById('user-greeting').innerText = `Chào ${user}!`;
        } else {
            alert("Em hãy nhập tên để HLV biết là ai nhé!");
        }
    }

    // 2. Cấu hình Nhận diện giọng nói (Web Speech API)
    const recognition = window.SpeechRecognition || window.webkitSpeechRecognition ? new (window.SpeechRecognition || window.webkitSpeechRecognition)() : null;
    if (recognition) {
        recognition.lang = 'vi-VN';
        recognition.continuous = true;
        recognition.interimResults = true;
    }

    let startTime;
    let finalTranscript = "";

    function startCoach() {
        if (!recognition) return alert("Trình duyệt của em không hỗ trợ nhận diện giọng nói.");
        
        finalTranscript = "";
        document.getElementById('output-text').innerText = "Đang ghi âm...";
        document.getElementById('ai-result').classList.add('hidden');
        document.getElementById('visualizer').classList.add('recording');
        
        document.getElementById('start-btn').classList.add('hidden');
        document.getElementById('stop-btn').classList.remove('hidden');

        startTime = Date.now();
        recognition.start();

        recognition.onresult = (event) => {
            let interim = "";
            for (let i = event.resultIndex; i < event.results.length; ++i) {
                if (event.results[i].isFinal) finalTranscript += event.results[i][0].transcript;
                else interim += event.results[i][0].transcript;
            }
            document.getElementById('output-text').innerText = finalTranscript || interim;
        };
    }

    function stopCoach() {
        recognition.stop();
        document.getElementById('visualizer').classList.remove('recording');
        document.getElementById('start-btn').classList.remove('hidden');
        document.getElementById('stop-btn').classList.add('hidden');
        
        // Chờ một chút để xử lý văn bản cuối cùng
        setTimeout(runAIAnalysis, 800);
    }

    // 3. "Bộ não" AI Phân tích phong thái
    function runAIAnalysis() {
        const text = document.getElementById('output-text').innerText.toLowerCase();
        const duration = (Date.now() - startTime) / 1000; // giây
        
        // Danh sách các từ ngập ngừng (ậm ờ)
        const fillers = ['à', 'ờ', 'ừm', 'uhm', 'thì', 'là', 'kiểu như'];
        let fillerCount = 0;
        fillers.forEach(f => {
            const matches = text.match(new RegExp(`\\b${f}\\b`, 'g'));
            if (matches) fillerCount += matches.length;
        });

        // Phân tích tốc độ (Số từ trên phút - WPM)
        const wordCount = text.split(' ').length;
        const wpm = Math.round((wordCount / duration) * 60);

        // Hiển thị kết quả
        document.getElementById('ai-result').classList.remove('hidden');
        document.getElementById('res-fillers').innerText = `${fillerCount} lần`;
        
        // Logic nhận xét phong thái
        if (fillerCount === 0) {
            document.getElementById('res-status').innerText = "Rất chuyên nghiệp, trôi chảy!";
            document.getElementById('res-advice').innerText = "Phong thái tuyệt vời, hãy phát huy!";
        } else if (fillerCount < 4) {
            document.getElementById('res-status').innerText = "Khá tự tin nhưng còn hơi rung.";
            document.getElementById('res-advice').innerText = "Hãy hít sâu trước khi bắt đầu để giảm từ 'à, ờ'.";
        } else {
            document.getElementById('res-status').innerText = "Bị ngập ngừng nhiều.";
            document.getElementById('res-advice').innerText = "Em nên soạn ý chính ra giấy để tránh bị bí từ khi nói.";
        }

        // Nhận xét tốc độ
        let speedText = "Vừa phải";
        if (wpm > 160) speedText = "Hơi nhanh (Cần nói chậm lại)";
        else if (wpm < 80) speedText = "Hơi chậm (Cần năng động hơn)";
        document.getElementById('res-speed').innerText = speedText;
    }
</script>
